const int mod = 1e9 + 7 ; 
signed main() {
    
    int a[10] = {1,1,1,2,2,3,4,5,5,5};
    vector<pair<int,int>> vi;
    int prev = 0;
    for(int i=0;i<10;i++) {
        while(i+1 < 10 && a[i] == a[i+1]) {
            i++;
        }
        
        vi.push_back(make_pair(a[i],i+1-prev));
        prev = i+1;
    }
                     
    for(auto it : vi) {
        cout<<it.first<<" "<<it.second<<endl;
    }
    
}
